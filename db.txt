  field name text;
  field id int rowid;
  comment "Controlling organisation.";
}

  field company struct cid:company.id;
  field cid;
  field email text unique;
  field name text comment "Full name.";
  field uid int rowid;
  list cid;

  update cid: uid;
  comment "A regular user.";
}
