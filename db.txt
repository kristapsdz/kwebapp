struct company {
  field name text;
  field id int rowid;
}

struct user {
  field company struct cid:company.id;
  field cid;
  field email text unique;
  field name text;
  field uid int rowid;
  comment 
	"A regular user.
	 Comments on a new line.";
  list cid;
  iterate name;
}

struct session { 
  field user struct userid:user.uid;
  field company struct cid:company.id;
  field cid int;
  field userid int comment "Associated user.";
  field token int comment "Random cookie.";
  field id int rowid;
  comment "Authenticated session.";
  search user.company.name,id: name foo comment "Search by foos";
}
